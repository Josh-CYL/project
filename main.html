        window.onload = Reset;
        window.onmousemove = Reset;
        window.onkeypress = Reset;
        window.onmousedown = Reset; // 點擊螢幕
        window.onclick = Reset;     // 按按鍵
        window.onscroll = Reset;    // 滑動
        function logout() {
           window.location = "/index.html";
        }

        function Reset() {
           clearTimeout(time);
           var minutes = 10; //minutes
           var timeout = parseInt(minutes) * 60 * 1000; //1s = 1000ms
           time = setTimeout(logout, timeout);
        }
      };
